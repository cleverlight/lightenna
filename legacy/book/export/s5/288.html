4.1.12. Exception handling

  • DECLARE
    • invalid_age exception;
  • BEGIN
    • IF (age < 0) THEN
      • RAISE invalid_age
    • END IF;
  • EXCEPTION
    • WHEN invalid_age THEN
      • INSERT INTO temp VALUES( ‘Cannot have negative ages’);
    • END;
